from itertools import combinations
from math import sqrt, floor

def solution(nums):
    cnt = 0
    for comb in combinations(nums, 3):
        temp = sum(comb)
        flag = True
        for i in range(2, floor(sqrt(temp))+1):
            if not (temp%i): flag = False
        if flag: cnt += 1
    return cnt
