def solution(prices):
    answer, n = [], len(prices)
    for i in range(n-1):
        flag = True
        for j in range(i+1, n):
            if prices[j] < prices[i]:
                answer.append(j-i)
                flag = False
                break
        if flag: answer.append(n-1-i)
    return answer + [0]
