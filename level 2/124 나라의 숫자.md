def solution(n):
    ternary = ''
    ternElements = {'1':'0','2':'1','4':'2'}
    while (n):
        ternary += str(n%3)
        n //= 3
    ternary = ternary[::-1]
    while ternary.count('0'):
        for i in range(len(ternary)-1):
            if ternary[i+1] == '0':
                ternary = str(int(ternary[:i] + ternElements[ternary[i]] + '4' + ternary[i+2:]))
                break
    return ternary

def solution(n):
    res = ''
    while n:
        if not n%3:
            res += '4'
            n -= 1
        else: res += str(n%3)
        n //= 3
    return res[::-1]
