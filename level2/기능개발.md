from math import ceil
def solution(prog, speed):
    time = []
    for i in range(len(prog)): time.append(ceil((100-prog[i])/speed[i]))
    init = time[0]
    cnt = 1
    res = []
    for j in range(1, len(time)):
        if time[j] <= init: cnt += 1
        else:
            res.append(cnt)
            init = time[j]
            cnt = 1
    res.append(cnt)
    return res
